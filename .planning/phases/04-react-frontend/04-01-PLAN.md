4. **Create `client/src/api/useApi.js`** — direct React port of prototype's `api.js` per D-05. Pattern from RESEARCH.md (Pattern 1):
   ```jsx
   import { useContext, useCallback } from 'react';
   import { AuthContext } from '../auth/AuthContext';
   const API_BASE = '/api';
   export function useApi() {
     const { token, logout } = useContext(AuthContext);
     const request = useCallback(async (path, { method = 'GET', body, headers = {} } = {}) => {
       const finalHeaders = { 'Content-Type': 'application/json', ...headers };
       if (token) finalHeaders['Authorization'] = `Bearer ${token}`;
       const res = await fetch(`${API_BASE}${path}`, {
         method, headers: finalHeaders,
         body: body !== undefined ? JSON.stringify(body) : undefined,
       });
       if (res.status === 401) { logout(); throw new Error('Session expired'); }
       if (!res.ok) {
         let detail = res.statusText;
         try { const data = await res.json(); detail = data.error || data.detail || detail; } catch {}
         const err = new Error(`${res.status} ${detail}`); err.status = res.status; throw err;
       }
       if (res.status === 204) return null;
       const ct = res.headers.get('content-type') || '';
       return ct.includes('application/json') ? res.json() : res.text();
     }, [token, logout]);
     return {
       get: (path) => request(path),
       post: (path, body) => request(path, { method: 'POST', body }),
       put: (path, body) => request(path, { method: 'PUT', body }),
       patch: (path, body) => request(path, { method: 'PATCH', body }),
       del: (path) => request(path, { method: 'DELETE' }),
     };
   }
   ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → Express API | Untrusted input crosses here (login, register, setup-password forms) |
| localStorage → AuthContext | Token retrieval from client-side storage; vulnerable to XSS |
| ProtectedRoute → /login | Route redirection when no valid auth |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-01 | Spoofing | LoginPage → /api/auth/login | mitigate | Use HTTPS in production; server validates credentials with bcrypt |
| T-04-02 | Tampering | XHR requests to API | mitigate | useApi hook adds JWT Bearer token; server validates token via requireAuth middleware |
| T-04-03 | Info Disclosure | localStorage token access | accept | D-10 mandates localStorage; XSS mitigation via React auto-escaping; no third-party scripts |
| T-04-04 | Elevation of Privilege | AdminRoute bypass | mitigate | AdminRoute checks `user.role === 'admin'`; server-side requireRole middleware also enforces |
| T-04-05 | Info Disclosure | Error messages in toast | mitigate | useApi extracts server error messages (already sanitized by Express error handler); no stack traces shown |
</threat_model>
